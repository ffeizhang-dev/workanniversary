## 🎨 随机生成逻辑

### 智能匹配算法

```javascript
function generateBlessingMessage(employee) {
    const { name, years, department, position, joinDate } = employee;

    // 1. 根据年限选择模板池
    let templatePool = [];

    if (years === 1) {
        templatePool = [...firstYearTemplates, ...generalTemplates];
    } else if (years === 3) {
        templatePool = [...thirdYearTemplates, ...generalTemplates];
    } else if (years === 5) {
        templatePool = [...fifthYearTemplates, ...generalTemplates];
    } else if (years === 10) {
        templatePool = [...tenthYearTemplates, ...generalTemplates];
    } else {
        templatePool = generalTemplates;
    }

    // 2. 根据部门优先选择专属模板（可选）
    const deptTemplates = getDepartmentTemplates(department);
    if (deptTemplates.length > 0 && Math.random() > 0.5) {
        templatePool = [...deptTemplates, ...templatePool];
    }

    // 3. 随机选择一个模板
    const randomIndex = Math.floor(Math.random() * templatePool.length);
    const template = templatePool[randomIndex];

    // 4. 替换变量
    let message = template.content
        .replace(/{name}/g, name)
        .replace(/{years}/g, years)
        .replace(/{department}/g, department)
        .replace(/{position}/g, position)
        .replace(/{joinDate}/g, formatDate(joinDate));

    // 5. 添加发送者签名
    message += `\n\n—— 小周 · ${new Date().toLocaleDateString('zh-CN')}`;

    return {
        message: message,
        templateId: template.id,
        style: template.style,
        milestone: template.milestone
    };
}
```

## 🛠️ 实现代码示例

### 完整功能代码

```javascript
// 祝福语生成和确认系统
class AnniversaryBlessingSystem {
    constructor() {
        this.templates = this.loadTemplates();
        this.pendingBlessings = [];
    }

    // 加载所有模板
    loadTemplates() {
        return {
            general: generalTemplates,
            year1: firstYearTemplates,
            year3: thirdYearTemplates,
            year5: fifthYearTemplates,
            year10: tenthYearTemplates,
            tech: techTemplates,
            product: productTemplates,
            operations: operationsTemplates
        };
    }

    // 检查今日周年员工
    checkTodayAnniversaries() {
        const today = new Date();
        const todayStr = `${today.getMonth() + 1}-${today.getDate()}`;

        const anniversaryEmployees = employees.filter(emp => {
            const joinDate = new Date(emp.joinDate);
            const empDateStr = `${joinDate.getMonth() + 1}-${joinDate.getDate()}`;
            return empDateStr === todayStr && emp.reminderEnabled;
        });

        return anniversaryEmployees.map(emp => ({
            ...emp,
            years: today.getFullYear() - new Date(emp.joinDate).getFullYear()
        }));
    }

    // 为员工生成祝福语
    generateBlessing(employee) {
        const { name, years, department, position, joinDate } = employee;

        // 选择合适的模板池
        let templatePool = this.templates.general;

        if (years === 1) {
            templatePool = [...this.templates.year1, ...this.templates.general];
        } else if (years === 3) {
            templatePool = [...this.templates.year3, ...this.templates.general];
        } else if (years === 5) {
            templatePool = [...this.templates.year5, ...this.templates.general];
        } else if (years === 10) {
            templatePool = [...this.templates.year10, ...this.templates.general];
        }

        // 随机选择
        const template = templatePool[Math.floor(Math.random() * templatePool.length)];

        // 替换变量
        let message = template.content
            .replace(/{name}/g, name)
            .replace(/{years}/g, years)
            .replace(/{department}/g, department)
            .replace(/{position}/g, position)
            .replace(/{joinDate}/g, this.formatDate(joinDate));

        message += `\n\n—— 小周 · ${new Date().toLocaleDateString('zh-CN')}`;

        return {
            employee: employee,
            message: message,
            templateStyle: template.style,
            generatedAt: new Date(),
            status: 'pending' // pending, confirmed, sent, cancelled
        };
    }

    // 生成今日所有祝福
    generateTodayBlessings() {
        const todayEmployees = this.checkTodayAnniversaries();

        if (todayEmployees.length === 0) {
            console.log('✅ 今天没有员工周年');
            return [];
        }

        this.pendingBlessings = todayEmployees.map(emp =>
            this.generateBlessing(emp)
        );

        console.log(`🎉 今天有 ${this.pendingBlessings.length} 位员工周年`);
        return this.pendingBlessings;
    }

    // 重新生成某个祝福语
    regenerateBlessing(employeeId) {
        const index = this.pendingBlessings.findIndex(
            b => b.employee.id === employeeId
        );

        if (index !== -1) {
            const employee = this.pendingBlessings[index].employee;
            this.pendingBlessings[index] = this.generateBlessing(employee);
            return this.pendingBlessings[index];
        }

        return null;
    }

    // 确认并发送
    async confirmAndSend(employeeId, customMessage = null) {
        const blessing = this.pendingBlessings.find(
            b => b.employee.id === employeeId
        );

        if (!blessing) {
            throw new Error('未找到待发送的祝福');
        }

        const messageToSend = customMessage || blessing.message;

        try {
            // 发送到飞书
            await this.sendToFeishu(blessing.employee, messageToSend);

            blessing.status = 'sent';
            blessing.sentAt = new Date();

            console.log(`✅ 已发送祝福给 ${blessing.employee.name}`);
            return true;
        } catch (error) {
            console.error('❌ 发送失败:', error);
            return false;
        }
    }

    // 发送到飞书
    async sendToFeishu(employee, message) {
        const feishuConfig = JSON.parse(localStorage.getItem('feishuConfig') || '{}');

        if (!feishuConfig.webhookUrl) {
            throw new Error('未配置飞书Webhook');
        }

        const card = {
            msg_type: 'interactive',
            card: {
                header: {
                    title: {
                        content: `🎉 ${employee.name}入职${employee.years}周年纪念`,
                        tag: 'plain_text'
                    },
                    template: 'turquoise'
                },
                elements: [
                    {
                        tag: 'div',
                        text: {
                            content: message,
                            tag: 'lark_md'
                        }
                    },
                    {
                        tag: 'hr'
                    },
                    {
                        tag: 'div',
                        text: {
                            content: `**员工信息**\n部门：${employee.department}\n职位：${employee.position}\n入职日期：${employee.joinDate}\n周年数：${employee.years}周年`,
                            tag: 'lark_md'
                        }
                    }
                ]
            }
        };

        const response = await fetch(feishuConfig.webhookUrl, {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json'
            },
            body: JSON.stringify(card)
        });

        const result = await response.json();

        if (result.code !== 0) {
            throw new Error(`飞书发送失败: ${result.msg}`);
        }
    }

    // 批量确认并发送
    async confirmAndSendAll() {
        const results = [];

        for (const blessing of this.pendingBlessings) {
            if (blessing.status === 'pending') {
                const success = await this.confirmAndSend(blessing.employee.id);
                results.push({
                    employee: blessing.employee.name,
                    success: success
                });
            }
        }

        return results;
    }

    // 格式化日期
    formatDate(dateStr) {
        const date = new Date(dateStr);
        return `${date.getFullYear()}年${date.getMonth() + 1}月${date.getDate()}日`;
    }

    // 发送确认请求到飞书HR群
    async sendConfirmationToHR() {
        const blessings = this.generateTodayBlessings();

        if (blessings.length === 0) {
            return;
        }

        const feishuConfig = JSON.parse(localStorage.getItem('feishuConfig') || '{}');
        const hrWebhook = feishuConfig.hrConfirmWebhook || feishuConfig.webhookUrl;

        for (const blessing of blessings) {
            const card = this.createConfirmationCard(blessing);

            await fetch(hrWebhook, {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json'
                },
                body: JSON.stringify(card)
            });
        }
    }

    // 创建确认卡片
    createConfirmationCard(blessing) {
        return {
            msg_type: 'interactive',
            card: {
                header: {
                    title: {
                        content: '🔔 待确认：今日周年祝福',
```

### 设置定时任务

#### 方法1：浏览器定时（简单）

```javascript
// 每天早上9点检查
function scheduleDaily Check() {
    const now = new Date();
    const target = new Date();
    target.setHours(9, 0, 0, 0);

    if (now > target) {
        target.setDate(target.getDate() + 1);
    }

    const timeUntilTarget = target - now;

    setTimeout(() => {
        blessingSystem.sendConfirmationToHR();
        scheduleDailyCheck(); // 递归设置下一次
    }, timeUntilTarget);
}

scheduleDailyCheck();
```
